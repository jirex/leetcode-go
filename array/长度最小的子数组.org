  package main

  import (
      "fmt"
      "math"
  )

  func main() {
      fmt.Println(minSubArrayLen(7, []int{2, 3, 1, 2, 4, 3}))
  }

  func minSubArrayLen(target int, nums []int) int {
      n := len(nums)

      if n == 0 {
          return 0
      }

      s, e := 0, 0

      ans := math.MaxInt32

      sum := 0

      for e < n {
          sum += nums[e]

          for sum >= target {
              ans = min(ans, e-s+1)
              sum -= nums[s]
              s++
          }
          e++
      }

      if ans == math.MaxInt32 {
          return 0
      }

      return ans

  }

  func min(x, y int) int {
      if x < y {
          return x
      }
      return y
  }
